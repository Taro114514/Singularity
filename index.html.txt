    <script>
        // YouTube再生関数（iPad用にバグを完全に修正しました）
        function loadYouTube() {
            const input = document.getElementById('ytUrl').value.trim();
            let videoId = input;

            if (input.includes('v=')) {
                videoId = input.split('v=')[1].split('&')[0];
            } else if (input.includes('youtu.be/')) {
                videoId = input.split('youtu.be/')[1].split('?')[0];
            }

            if (videoId) {
                const player = document.getElementById('ytPlayer');
                player.src = `https://youtube.com${videoId}`;
            } else {
                alert('正しいURLまたは動画IDを入力してください。');
            }
        }

        // YouTubeお気に入り用
        function playFav(id) {
            document.getElementById('ytUrl').value = id;
            loadYouTube();
        }

        // Xの表示アカウントを切り替える関数
        function changeX(screenName) {
            const container = document.getElementById('xContainer');
            container.innerHTML = `<a class="twitter-timeline" href="https://twitter.com{screenName}?ref_src=twsrc%5Etfw">Tweets by ${screenName}</a>`;
            
            if (window.twttr && window.twttr.widgets) {
                window.twttr.widgets.load(container);
            }
        }
    </script>
